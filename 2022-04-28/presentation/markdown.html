library(tidyverse)
df <- data.frame(
  x = c(1:100),
  y = runif(100),
  class = sample(c('0', '1'), 
                 size = 100, 
                 replace = T)
)
df %>% ggplot() +
  aes(x = x, y = y, colour = class) +
  geom_point() +
  labs(title = 'My important insights')
